create table public.kunden (
  id                  integer generated by default as identity primary key,
  ist_kunde           boolean not null default true,   -- aus IstKunde
  kundennr            text unique,                     -- KundenNr (AS-xxx etc.)

  firma               text,
  nachname            text,
  vorname             text,
  titel               text,
  anrede              text,

  strasse             text,
  hausnr              text,
  laenderkennung      varchar(5),
  plz                 varchar(10),
  ort                 text,
  ortsteil            text,

  telefonnr           text,
  telefonnr_geschaeft text,
  mobilnr             text,
  mobilnr2            text,
  faxnr               text,
  email               text,
  homepage            text,

  comment             text,
  interne_anmerkungen text,
  bezeichnung         text,

  trace_mark          integer,
  last_update         timestamptz,
  created_at          timestamptz not null default now()
);

  faxnr               text,
  email               text,
  homepage            text,

  comment             text,
  anmerkungen         text,
  last_update         timestamptz,

create table public.gemeinden (
  id                   integer primary key,           -- DefGemeindenID
  bezeichnung          text not null,
  typ                  text,
  funktion             text,
  schluessel           integer,                       -- war float, ist aber logisch ein Integer
  def_bundeslaender_id integer,
  comment              text,
  last_update          timestamptz,
  custom               integer not null default 0
);

create table public.anl_typen (
  id                        integer primary key,        -- DefAnlTypenID
  sortiernr                 integer,
  bezeichnung               text not null,
  bio_felder                text,                       -- die |...|-Kette aus BioFelder

  preis_je_wartung          numeric(10,2),
  preis_je_kontrolle        numeric(10,2),

  -- NEU: Wartungsstammdaten laut Anforderung
  wartungsintervall_monate  integer not null default 12,
  dauer_wartung_minuten     integer not null default 60,
  dauer_kontrolle_minuten   integer,

  comment                   text,
  last_update               timestamptz
);

create table public.anlagen (
  id                        integer primary key,        -- AnlID, aus Altbestand übernommen
  kunden_id                 integer not null
                               references public.kunden(id)
                               on update cascade on delete restrict,

  ist_aktiv                 boolean not null default true,   -- IstAktiv 0/1

  anlagen_nr                text unique,                     -- AnlagenNr
  verfahren_br_anz_behaelter integer,                        -- VerfahrenBRAnzBehaelter

  strasse                   text,
  hausnr                    text,
  laenderkennung            varchar(5),
  plz                       varchar(10),
  ort                       text,
  ortsteil                  text,

  gemarkung                 text,
  flur                      text,
  flurstueck                text,

  touren_nr                 integer,
  touren_nr2                integer,
  touren_nr3                integer,

  rechtswert                integer,
  hochwert                  integer,

  breitengrad               text,   -- im Bestand als "00°  00´  00,0´´"
  laengengrad               text,

  anlage_ausgelegt_ew       integer,          -- AnlageAusgelegtEW
  tatsaechliche_ew          integer,          -- TatsaechlicheEW
  gesamtgroesse_vk          numeric(12,2),    -- GesamtgroesseVK

  datum_naechste_wartung    date,
  datum_abgabefrei_seit     date,

  -- alte Vertragsfelder aus tblanl (können optional weitergeführt werden,
  -- die eigentlichen Verträge liegen künftig in wartungsvertraege)
  wartungsvertrag_flag      smallint,        -- aus Wartungsvertrag (1/2)
  datum_wartungsvertrag     date,

  export_erlaubt_wartung    boolean default true,   -- ExportErlaubtWartung

  comment                   text,
  anmerkungen_gesamt        text,

  last_update               timestamptz,
  last_export               timestamptz,
  last_import               timestamptz,
  trace_mark                integer,

  bezeichnung               text,

  -- NEU: Wartungskontakt
  kontakt_kunde_id          integer
                               references public.kunden(id)
                               on update cascade on delete set null,
  kontakt_id                integer
                               references public.kontakte(id)
                               on update cascade on delete set null,

  created_at                timestamptz not null default now(),

  -- Es darf höchstens EIN Kontakt gesetzt sein (oder keiner).
  constraint anlagen_kontakt_xor check (
    (kontakt_kunde_id is not null and kontakt_id is null)
    or (kontakt_kunde_id is null and kontakt_id is not null)
    or (kontakt_kunde_id is null and kontakt_id is null)
  )
);

create table public.wartungsvertraege (
  id                          integer generated by default as identity primary key,

  anlage_id                   integer not null
                                 references public.anlagen(id)
                                 on update cascade on delete cascade,

  kunde_id                    integer
                                 references public.kunden(id)
                                 on update cascade on delete set null,

  vertragsnummer              text,
  vertragsdatum               date not null,        -- z.B. aus altem DatumWartungsvertrag
  gueltig_ab                  date,
  gueltig_bis                 date,

  anl_typ_id                  integer
                                 references public.anl_typen(id)
                                 on update cascade on delete set null,

  intervall_monate            integer,              -- optional Override
  dauer_wartung_minuten       integer,              -- optional Override
  preis_je_wartung            numeric(10,2),        -- optional Override des Typs

  aktiv                       boolean not null default true,

  comment                     text,
  last_update                 timestamptz,
  created_at                  timestamptz not null default now()
);
